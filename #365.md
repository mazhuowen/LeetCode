class Solution {
    public boolean canMeasureWater(int x, int y, int z) {
        if (z == x || z == y || z == x + y) return true;
        int diff = diff(x, y);
        if (z == diff || z == diff + x || z == diff + y) return true;
        diff = diff(y, x);
        if (z == diff || z == diff + x || z == diff + y) return true;
        return false;
    }

    private int diff(int x, int y) {
        int res = y - x;
        while (res < 0) res += y;
        return res;
    }
}
