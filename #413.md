class Solution {
    public int numberOfArithmeticSlices(int[] A) {
        if (A == null || A.length < 3) return 0;

        int count = 0;
        int[] dp = new int[A.length];
        for (int i = 2; i < A.length; i++) {
            // 当前元素与前一区间构成等差数列
            if (A[i] - A[i - 1] == A[i - 1] - A[i - 2]) {
                // 以A[i]结尾的等差数列为当前3个值及与A[i-1]中的等差数列之和
                dp[i] = dp[i - 1] + 1;
            }
            count += dp[i];
        }
        return count;
    }
}
