class Solution {
    public int bulbSwitch(int n) {
        if (n <= 0) return 0;
        int res = 1;
        int count = 0;
        for (int i = 1; i <= n; i++) {
            if (++count == 2 * res + 1) {
                res++;
                count = 0;
            }
        }
        if (count == 0) res--;
        return res;
    }
}

class Solution {
    public int bulbSwitch(int n) {
        if (n <= 0) return 0;
        int count = 0;
        for (int i = 3; n > i; i += 2) {
            n -= i;
            count++;
        }
        return count + 1;
    }
}
