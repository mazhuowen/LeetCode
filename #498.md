class Solution {
    public int[] findDiagonalOrder(int[][] matrix) {
        if (matrix == null || matrix.length == 0 || matrix[0].length == 0) return new int[]{};

        int m = matrix.length, n = matrix[0].length;
        int[] res = new int[m * n];
        int i = 0, j = 0, idx = 0;
        while (i != m - 1 || j != n - 1) {
            res[idx++] = matrix[i][j];
            // 向上方向
            if ((i + j) % 2 == 0) {
                // 到达末端，转为向下方向
                if (j == n - 1) {
                    i++;
                } else if (i == 0) {
                    j++;
                }else {
                    i--;
                    j++;
                }
            }
            // 向下反向
            else {
                if (i == m - 1) {
                    j++;
                } else if (j == 0) {
                    i++;
                } else {
                    i++;
                    j--;
                }
            }
        }
        res[idx] = matrix[i][j];
        return res;
    }
}
