class Solution {
    public int singleNumber(int[] nums) {
        long mask = 1;
        int res = 0, count = 0, epoch = 0;

        while (epoch < 32) {
            for (int num : nums) {
                if ((num & mask) == mask) count++;
            }
            // 说明此位为1
            if (count % 3 == 1) res |= mask;
            mask <<= 1;
            epoch++;
            count = 0;
        }
        return res;
    }
}
