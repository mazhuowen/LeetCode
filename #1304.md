class Solution {
    public int[] sumZero(int n) {
        if (n <= 0) throw new IllegalArgumentException("invalid param");
        int[] res = new int[n];

        int base = 1;
        int idx = n / 2 - 1;
        while (idx >= 0) {
            res[idx] = -base;
            res[n - idx - 1] = base;
            idx--;
            base++;
        }
        return res;
    }
}
