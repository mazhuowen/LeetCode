class Solution {
    public boolean isOneBitCharacter(int[] bits) {
        if (bits == null || bits.length == 0) return false;

        int n = bits.length;
        if (bits[n - 1] == 1) return false;
        if (bits.length == 1 || bits[bits.length - 2] == 0) return true;


        int count = 0;
        for (int i = n - 2; i >= 0 && bits[i] == 1; i--) {
            count++;
        }
        return count % 2 == 0;
    }
}
