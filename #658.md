class Solution {
    public List<Integer> findClosestElements(int[] arr, int k, int x) {
        if (k <= 0) throw new IllegalArgumentException("invalid param");
        LinkedList<Integer> res = new LinkedList<>();
        if (k >= arr.length) {
            for (int num : arr) res.addLast(num);
            return res;
        }

        int left = 0, right = arr.length - 1;
        while (left < right) {
            int mid = left + (right - left) / 2;
            if (arr[mid] >= x) right = mid;
            else left = mid + 1;
        }
        // left为绝对值最小的索引
        if (left > 0 && arr[left] > x) {
            if (Math.abs(arr[left - 1] - x) < Math.abs(arr[left] - x)) {
                left--;
            }
        }
        // 以left为中心扩展
        int i = left - 1, j = left;
        while ((i >= 0 || j < arr.length) && j - i <= k) {
            if (i >= 0 && Math.abs(arr[i] - x) <= Math.abs(arr[j] - x) || j >= arr.length) {
                res.addFirst(arr[i--]);
            } else {
                res.addLast(arr[j++]);
            }
        }
        return res;
    }
}
