class Solution {
    public int numTimesAllBlue(int[] light) {
        int sum1 = 0, sum2 = 0;
        int count = 0;
        for (int i = 0; i < light.length; i++) {
            sum1 += i + 1;
            sum2 += light[i];
            if (sum1 == sum2)count++;
        }
        return count;
    }
}
