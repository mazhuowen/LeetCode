class Solution {
    public String reverseVowels(String s) {
        String seq = "aeiouAEIOU";
        char[] str = s.toCharArray();
        int left = 0, right = str.length - 1;
        while (left < right) {
            if (!seq.contains(str[left] + "")) left++;
            else if (!seq.contains(str[right] + "")) right--;
            else {
                char temp = str[left];
                str[left++] = str[right];
                str[right--] = temp;
            }
        }
        return new String(str);
    }
}
