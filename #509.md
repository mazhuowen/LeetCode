class Solution {
    int size = 1;
    int[] fib = new int[31];
    {
        fib[0] = 0;
        fib[1] = 1;
    }

    public int fib(int N) {
        if (N <= size) return fib[N];

        for (int i = size + 1; i <= N; i++) {
            fib[i] = fib[i - 1] + fib[i - 2];
        }
        size = N;
        return fib[N];
    }
}
