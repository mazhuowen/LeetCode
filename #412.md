class Solution {
    public List<String> fizzBuzz(int n) {
        List<String> res = new ArrayList<>(n);
        for (int i = 0; i < n; i++) {
            res.add(Integer.toString(i + 1));
        }

        for (int i = 3; i <= n; i += 3) {
            res.set(i - 1, "Fizz");
        }
        for (int i = 5; i <= n; i += 5) {
            res.set(i - 1, "Buzz");
        }
        for (int i = 15; i <= n; i += 15) {
            res.set(i - 1, "FizzBuzz");
        }
        return res;
    }
}
