class Solution {
    public List<Integer> getRow(int rowIndex) {
        int[] dp = new int[rowIndex + 1];
        // 初始化第一行
        dp[0] = 1;
        for (int i = 1; i <= rowIndex; i++) {
            for (int j = i - 1; j >= 1; j--) {
                dp[j] += dp[j - 1];
            }
            // 末尾添加1
            dp[i] = 1;
        }
        List<Integer> res = new ArrayList<>();
        for (int num : dp) res.add(num);
        return res;
    }
}
